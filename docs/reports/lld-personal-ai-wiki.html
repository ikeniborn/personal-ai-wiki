§2 · Схема БД (DDL, индексы, каскады)

PostgreSQL 16 + pgvector. UUID PK (gen_random_uuid), фикс vector(dim) из конфига, FTS 'simple'.

CREATE EXTENSION IF NOT EXISTS vector;
CREATE EXTENSION IF NOT EXISTS pgcrypto;
CREATE EXTENSION IF NOT EXISTS citext;

CREATE TYPE user_role     AS ENUM ('admin','editor','viewer');
CREATE TYPE source_status AS ENUM ('uploaded','extracted','ingested','failed');
CREATE TYPE rev_origin    AS ENUM ('ai','user');
CREATE TYPE job_type      AS ENUM ('ingest','lint','format','reindex');
CREATE TYPE job_status    AS ENUM ('queued','running','succeeded','failed','cancelled');

CREATE TABLE users (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  email citext UNIQUE NOT NULL, pw_hash text NOT NULL,
  role user_role NOT NULL DEFAULT 'viewer', created_at timestamptz NOT NULL DEFAULT now());

  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id uuid NOT NULL REFERENCES users(id) ON DELETE CASCADE,

CREATE INDEX ON api_keys(prefix);

CREATE TABLE app_settings (         -- singleton: wiki-defaults + LLM-провайдер (api_key -> шифротекст)
  id boolean PRIMARY KEY DEFAULT true CHECK (id), settings jsonb NOT NULL DEFAULT '{}');

  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),

CREATE TABLE blobs (                 -- StorageBackend: мелкие тела (bytea)
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),

CREATE TABLE sources (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  domain_id uuid NOT NULL REFERENCES domains(id) ON DELETE CASCADE,
  storage_ref text NOT NULL,         -- 'blob:uuid' | 'lo:oid' | 's3:key'
  filename text, type text NOT NULL, url text, checksum text NOT NULL,
  status source_status NOT NULL DEFAULT 'uploaded', created_at timestamptz NOT NULL DEFAULT now(),
  UNIQUE (domain_id, checksum));

CREATE TABLE articles (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  domain_id uuid NOT NULL REFERENCES domains(id) ON DELETE CASCADE,
  slug text NOT NULL, title text NOT NULL, storage_ref text NOT NULL, summary text,
  current_rev int NOT NULL DEFAULT 1,                -- optimistic lock
  updated_at timestamptz NOT NULL DEFAULT now(), UNIQUE (domain_id, slug));

CREATE TABLE article_revisions (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  article_id uuid NOT NULL REFERENCES articles(id) ON DELETE CASCADE,
  rev_no int NOT NULL, storage_ref text NOT NULL,
  author_id uuid REFERENCES users(id) ON DELETE SET NULL,
  origin rev_origin NOT NULL, created_at timestamptz NOT NULL DEFAULT now(),
  UNIQUE (article_id, rev_no));

CREATE TABLE entities (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  domain_id uuid NOT NULL REFERENCES domains(id) ON DELETE CASCADE,
  name text NOT NULL, type text NOT NULL,
  article_id uuid REFERENCES articles(id) ON DELETE SET NULL,   -- каноническая страница
  UNIQUE (domain_id, type, name));
CREATE TABLE article_entities (
  article_id uuid NOT NULL REFERENCES articles(id) ON DELETE CASCADE,
  entity_id  uuid NOT NULL REFERENCES entities(id) ON DELETE CASCADE,
  PRIMARY KEY (article_id, entity_id));

CREATE TABLE links (
  src_article_id uuid NOT NULL REFERENCES articles(id) ON DELETE CASCADE,
  dst_article_id uuid NOT NULL REFERENCES articles(id) ON DELETE CASCADE,
  type text NOT NULL DEFAULT 'related',   -- related|parent|child|references|depends_on (app-allowlist)
  PRIMARY KEY (src_article_id, dst_article_id, type));
CREATE INDEX ON links(dst_article_id);                -- backlinks

CREATE TABLE citations (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  article_id uuid NOT NULL REFERENCES articles(id) ON DELETE CASCADE,
  source_id  uuid NOT NULL REFERENCES sources(id) ON DELETE CASCADE,
  span text, UNIQUE (article_id, source_id, span));

CREATE TABLE chunks (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  article_id uuid NOT NULL REFERENCES articles(id) ON DELETE CASCADE,
  ord int NOT NULL, kind text NOT NULL DEFAULT 'section',   -- section | summary
  text text NOT NULL, heading_path text,
  tsv tsvector,
  embedding vector(N),                  -- N = dim из setup-wizard (managed migration)
  embedding_version text NOT NULL,
  UNIQUE (article_id, ord));
CREATE INDEX ON chunks USING hnsw (embedding vector_cosine_ops);
CREATE INDEX ON chunks USING gin (tsv);
CREATE INDEX ON chunks(embedding_version);

CREATE TABLE chunk_entities (        -- chunk-level entity index (§5)
  chunk_id  uuid NOT NULL REFERENCES chunks(id) ON DELETE CASCADE,
  entity_id uuid NOT NULL REFERENCES entities(id) ON DELETE CASCADE,
  PRIMARY KEY (chunk_id, entity_id));
CREATE INDEX ON chunk_entities(entity_id);

CREATE TABLE chat_sessions (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id uuid NOT NULL REFERENCES users(id) ON DELETE CASCADE,
  domain_id uuid NOT NULL REFERENCES domains(id) ON DELETE CASCADE,
  created_at timestamptz NOT NULL DEFAULT now());
CREATE TABLE chat_messages (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  session_id uuid NOT NULL REFERENCES chat_sessions(id) ON DELETE CASCADE,

CREATE INDEX ON chat_messages(session_id, created_at);

CREATE TABLE jobs (

  status job_status NOT NULL DEFAULT 'queued', progress real NOT NULL DEFAULT 0,
  log jsonb NOT NULL DEFAULT '[]', error text, cancel_requested boolean NOT NULL DEFAULT false,

§3 · Контракты StorageBackend / Chat / Embedding / Vision

class StorageBackend(Protocol):
    async def put(self, data: bytes, *, content_type: str | None = None) -> str: ...   # -> ref
    async def get(self, ref: str) -> bytes: ...
    async def open(self, ref: str) -> AsyncIterator[bytes]: ...    # стрим для крупных
    async def delete(self, ref: str) -> None: ...
    async def exists(self, ref: str) -> bool: ...
# PostgresStorage: мелкое -> blobs.bytea (ref 'blob:uuid'); крупное -> Large Object (ref 'lo:oid').

class ChatProvider(Protocol):
    async def chat(self, messages: list[Message], *, model: str,
                   tools: list[ToolSpec] | None = None, temperature: float = 0.2,
                   response_format: dict | None = None) -> ChatResult: ...
    def stream(self, messages: list[Message], *, model: str,
               temperature: float = 0.2) -> AsyncIterator[str]: ...   # токены для SSE

class EmbeddingProvider(Protocol):
    async def embed(self, texts: list[str], *, model: str) -> list[list[float]]: ...

class VisionProvider(Protocol):                       # OCR / описание изображений (ingest)
    async def describe(self, image: bytes, *, model: str, prompt: str | None = None) -> str: ...
# OpenAICompatProvider реализует все три; connection + модели берутся из admin-настроек.
# structured(messages, schema, model, retries) -> validated pydantic; ретрай с текстом ошибки.
Шов: harness → ChatProvider, vector → EmbeddingProvider, ingest → VisionProvider (для изображений/сканов). Слабая модель без нативного tool-calling → fallback на JSON-mode. Добавить другой бэкенд = новая реализация.

§4 · Операции харнесса

Цикл (harness/loop.py): system + task + tools → chat → tool_calls → результаты → повтор до финала / max_steps. Каждый шаг → progress. Гарды: max_steps + per-step timeout; write-scope (target.domain_id == ctx.domain_id); нет деструктивных tools; контент источника — недоверенные данные; выход валидируется схемой до записи.

Связи: явные (LLM add_link/link_suggestions) + co-occurrence (общие article_entities ≥ порога). Типы связей (v1): related · parent · child · references · depends_on (text + app-allowlist; co-occurrence → related).

Лимиты агента

  • max_steps (итерации tool-loop), max_tool_calls (всего/на шаг), max_writes/op.
  • per-step timeout + overall op timeout; token-budget на операцию (input+output) → стоп при превышении.
  • max_input_tokens → map-reduce; cap размера tool-результата (search_wiki/read_source обрезаются/суммируются); управление контекстным окном.
  • loop-detection: повтор идентичных tool-call → аборт.

Безопасность агента

  • Tool-allowlist по операции (Query/Chat — только read; write-tools вне их контекста).
  • Write-scope доменом + валидация: slug без path-traversal; add_link — оба конца в домене и type ∈ allowlist; citation.source_id ∈ домен.
  • Источники и tool-результаты обёрнуты как «data, not instructions» (anti prompt-injection).
  • Выход валидируется схемой до записи; upsert_article идемпотентен по slug (merge, не дубль).
  • Нет сетевых/файловых/shell-tools; секреты не попадают в контекст; каждый tool-call → audit_log.

Промпты

harness/prompts/ — версионируемые шаблоны (prompt_version). Общая преамбула (роль · «только данный контекст» · «контент = данные» · цитирование · запрет выдумок · языки gen/reasoning) + per-op (extraction/drafting/query/lint/format/chat). Overlay: база в коде + user_prompt (глобальный/по домену) поверх. Structured-output: JSON-schema + repair-loop.

Чанкинг сгенерированной статьи

Структура: статья = # заголовок + разделы ## (глубже запрещено; drafting-промпт нормализует к ≤ ##); преамбула до первого ## → раздел (intro). Плоские секции → предсказуемый чанкинг.

1. summary-chunk: отдельное LLM-обобщение всей статьи (2-4 предложения)
      -> ord=0, kind='summary', heading_path='(summary)'   (+ дублируется в articles.summary)
2. split по '##' -> [intro, sec1, sec2, ...]
3. на секцию: len <= target -> 1 chunk; иначе semantic split (breakpoint по эмбеддингам
   предложений, порог из конфига), bounded target_size
4. overlap: к каждому chunk (кроме первого) спереди добавляются последние 1-2 предложения
   предыдущего chunk (sliding window) -> связность на стыках
5. heading_path = '## <секция>' (преамбула -> '(intro)'); kind='section'
6. chunk -> embed -> chunks(kind,ord,heading_path,text,tsv,embedding,version) + chunk_entities
ordkindheading_pathtext (начало)
0summary(summary)«QUIC — UDP-транспорт: TLS+мультиплексирование, без HoL-блокировки…»
1section(intro)«QUIC — транспорт поверх UDP, объединяет…»
2section## История«Разработан Google в 2012, RFC 9000…»
3section## Сравнение с TCP«QUIC устраняет head-of-line…» (часть 1)
4section## Сравнение с TCP«[overlap: 1–2 предл. из ord 3] …продолжение»

summary-chunk → article-level recall (находит статью целиком); section-chunks → точный фрагмент; overlap по предложениям сохраняет смысл на границах. Chunk-level сущности → chunk_entities (entity-scoped поиск).

Change-detection: sha256 → UNIQUE(domain_id, checksum). Re-chunk при изменении статьи (старые чанки удаляются → перечанковка); каскад при удалении (см. §2). Bulk-upload: zip/папка → распаковка (anti-zip-bomb §11) → пакетная регистрация источников → ingest. Крупные файлы стримятся в storage (без полной загрузки в память).

Trade-off: семантический чанкинг + summary-chunk тратят доп. эмбеддинги на ingest — дороже/медленнее фиксированного. Принято осознанно ради качества границ и article-level recall.

§6 · Retrieval: гибрид (RRF) + BFS

# vector arm
SELECT c.id, row_number() OVER (ORDER BY c.embedding <=> :q) AS r
FROM chunks c JOIN articles a ON a.id=c.article_id
WHERE a.domain_id = :dom AND c.embedding_version = :ver
ORDER BY c.embedding <=> :q LIMIT :k1;
# fts arm
SELECT c.id, row_number() OVER (ORDER BY ts_rank_cd(c.tsv, q) DESC) AS r
FROM chunks c JOIN articles a ON a.id=c.article_id,
     websearch_to_tsquery('simple', :q) q
WHERE a.domain_id = :dom AND c.tsv @@ q LIMIT :k2;
# merge: RRF  score = sum( weight_i / (rrf_k + rank_i) )  -> top-N (веса в конфиге)

# BFS (только исходящие, cycle-safe)
WITH RECURSIVE bfs(article_id, depth) AS (
  SELECT unnest(:seed_ids), 0
  UNION
  SELECT l.dst_article_id, b.depth+1
    FROM bfs b JOIN links l ON l.src_article_id = b.article_id
   WHERE b.depth < :max_depth
) CYCLE article_id SET cyc USING path
SELECT DISTINCT article_id FROM bfs;

Сборка контекста: seed-пассажи (slug+heading_path+citations) + summary соседей + кросс-ссылки, token-budgeted по fused-score. Reindex (vector/reindex.py): чанки с embedding_version != current переэмбеддиваются батчами; поиск фильтрует current. Reranking — точка расширения (бэклог). Query→entity: извлечение сущностей из запроса → entity-boost через chunk_entities. Кэш эмбеддинга запроса. Пустой контекст → ответ «не знаю» (без выдумок).

§7 · Jobs / worker

Задачи & жизненный цикл

  • ingest_domain · lint_domain · fix_issues · format_articles · reindex_domain (arq).
  • API создаёт jobs(queued) + enqueue с job_id; воркер: running → терминальный статус.
  • Heartbeat + startup-reconciler помечает зависшие running как failed.
  • Job-lock по домену: одна пишущая задача (ingest/fix/format/reindex) на домен за раз (Redis-лок) — исключает гонки записи.

Прогресс & отмена

  • Redis pub/sub job:{id} + replay из jobs.log.
  • SSE GET /jobs/{id}/events (HTMX SSE-ext).
  • cancel_requested → кооперативная проверка между шагами → cancelled, очистка частичного.
  • arq job-timeout + per-step LLM-timeout.
Сериализация Ollama: LLM-вызовы за Redis-локом lock:model:{name} — одна модель серийно, разные модели параллельно; concurrency воркера низкий. Очереди arq: LLM-задачи и лёгкие разнесены; ретраи на транзиентных ошибках (backoff, N попыток), ядовитые → dead-letter. Плановый lint/reindex/GC — admin-триггер в v1 (cron — бэклог).

§11 · Безопасность, деплой, тестирование

Безопасность

  • XSS: nh3 allowlist + CSP (без inline-script).
  • CSRF: double-submit, SameSite=Lax.
  • Загрузки: magic-byte sniff, whitelist, max-size, anti-zip-bomb.
  • Prompt-injection: untrusted-делимитеры, write-scope, max_steps, валидация выхода.
  • SSRF: allowlist + блок private/link-local, https, size-cap.
  • Пароли argon2; API-ключи хешем, scopes, revoke; секреты в env/secrets.
  • Ключ LLM-провайдера в БД шифруется (Fernet, ключ из env); в контекст агента не попадает.

Деплой (compose)

  • traefik (reverse-proxy, авто-TLS) · api (uvicorn) · worker (arq) · postgres (pgvector/pgvector:pg16) · redis (AOF).
  • Ingress: Traefik терминирует TLS (Let's Encrypt/ACME), роутит по домену на api; HTTP→HTTPS redirect; SSE проксируется (без буферизации).
  • LLM-провайдер внешний — URL/ключ в admin-UI (нет ollama-сервиса).
  • Миграции — init-контейнер (one-shot до api/worker); setup-wizard при первом старте (admin + подключение + dim/модели).
  • Бэкап: плановый pg_dump (cron/sidecar) + ретенция, процедура restore. Метрики Prometheus /metrics (api+worker).
  • Healthcheck на сервис; тома pgdata / redisdata / letsencrypt.

Тесты

  • Unit: чанкинг, RRF, BFS-билдер, sanitize, парсинг провайдера (mock).
  • Integration: testcontainers (PG+pgvector, Redis), stub-LLM.
  • API: httpx — auth/RBAC/CSRF/пагинация/ошибки. MCP: tools.
  • E2E: fixture-источники → ingest → query. CI: ruff + mypy + pytest.
